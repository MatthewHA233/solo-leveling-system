根据官方文档整理的关键信息：

- **WebSocket地址**：`wss://dashscope.aliyuncs.com/api-ws/v1/realtime?model=qwen3.5-omni-plus-realtime`
- **音频格式**：输入/输出均为 PCM，音频数据 Base64 编码
- **核心客户端事件**：`session.update`、`input_audio_buffer.append`、`input_audio_buffer.commit`
- **核心服务端事件**：`session.created`、`response.audio.delta`（含 Base64 音频）、`response.audio.done`、`response.text.delta`

以下是完整的 `qwen_omni.rs`：

```rust
// src-tauri/src/qwen_omni.rs
// Qwen3.5-Omni-Plus-Realtime 实时对话模块
// 协议参考：https://help.aliyun.com/zh/model-studio/realtime

use base64::{engine::general_purpose, Engine as _};
use futures_util::{SinkExt, StreamExt};
use serde::{Deserialize, Serialize};
use serde_json::{json, Value};
use std::sync::Arc;
use tauri::{AppHandle, Emitter};
use tokio::sync::{mpsc, Mutex};
use tokio_tungstenite::{connect_async, tungstenite::protocol::Message};
use uuid::Uuid;

// ── 常量 ──────────────────────────────────────────────────
const WS_URL_BASE: &str =
    "wss://dashscope.aliyuncs.com/api-ws/v1/realtime";
const MODEL: &str = "qwen3.5-omni-plus-realtime";

// PCM 参数（官方要求）
pub const SAMPLE_RATE: u32 = 16_000;
pub const CHANNELS: u16 = 1;
pub const BITS_PER_SAMPLE: u16 = 16;

// ── 服务端事件结构体（官方文档字段） ──────────────────────

/// session.created —— 连接成功后服务端第一个事件
#[derive(Debug, Deserialize)]
pub struct SessionCreated {
    pub event_id: String,
    #[serde(rename = "type")]
    pub event_type: String,
    pub session: Value,
}

/// response.audio.delta —— 增量音频数据（Base64 PCM）
#[derive(Debug, Deserialize)]
pub struct AudioDelta {
    pub event_id: String,
    #[serde(rename = "type")]
    pub event_type: String,
    pub response_id: String,
    pub item_id: String,
    pub output_index: u32,
    pub content_index: u32,
    pub delta: String, // Base64 编码的 PCM 数据
}

/// response.text.delta —— 增量文字
#[derive(Debug, Deserialize)]
pub struct TextDelta {
    pub event_id: String,
    #[serde(rename = "type")]
    pub event_type: String,
    pub delta: String,
}

// ── Tauri 前端事件载荷 ────────────────────────────────────

#[derive(Debug, Serialize, Clone)]
pub struct AudioChunkPayload {
    pub data: String, // Base64 PCM，前端直接解码播放
}

#[derive(Debug, Serialize, Clone)]
pub struct TextChunkPayload {
    pub text: String,
}

#[derive(Debug, Serialize, Clone)]
pub struct StatusPayload {
    pub status: String, // "connected" | "disconnected" | "error"
    pub message: Option<String>,
}

// ── 会话状态 ──────────────────────────────────────────────

pub struct OmniSession {
    /// 向 WebSocket 发送消息的通道
    tx: mpsc::UnboundedSender<Message>,
}

impl OmniSession {
    /// 向输入音频缓冲区追加 PCM 数据（input_audio_buffer.append）
    pub fn append_audio(&self, pcm_bytes: &[u8]) {
        let audio_b64 = general_purpose::STANDARD.encode(pcm_bytes);
        let msg = json!({
            "event_id": new_event_id(),
            "type": "input_audio_buffer.append",
            "audio": audio_b64
        });
        let _ = self.tx.send(Message::Text(msg.to_string().into()));
    }

    /// 手动模式下提交缓冲区（input_audio_buffer.commit）
    /// VAD 模式下无需调用，服务端自动提交
    pub fn commit_audio(&self) {
        let msg = json!({
            "event_id": new_event_id(),
            "type": "input_audio_buffer.commit"
        });
        let _ = self.tx.send(Message::Text(msg.to_string().into()));
    }

    /// 关闭 WebSocket
    pub fn disconnect(&self) {
        let _ = self.tx.send(Message::Close(None));
    }
}

// ── 全局会话（Tauri State） ───────────────────────────────

pub type SharedSession = Arc<Mutex<Option<OmniSession>>>;

// ── 核心：建立 WebSocket 连接 ─────────────────────────────

/// 连接到 Qwen-Omni-Realtime，完成 session.update 握手，
/// 并在后台持续接收服务端事件，通过 Tauri emit 推送给前端。
pub async fn connect(
    api_key: String,
    system_prompt: String,
    app: AppHandle,
    session_holder: SharedSession,
) -> Result<(), String> {
    // 1. 构造 WebSocket URL 和请求头
    let url = format!("{}?model={}", WS_URL_BASE, MODEL);
    let request = {
        use tokio_tungstenite::tungstenite::handshake::client::Request;
        use tokio_tungstenite::tungstenite::http::header::{
            AUTHORIZATION, HeaderValue,
        };
        let mut req = Request::builder()
            .uri(&url)
            .body(())
            .map_err(|e| e.to_string())?;
        req.headers_mut().insert(
            AUTHORIZATION,
            HeaderValue::from_str(&format!("Bearer {}", api_key))
                .map_err(|e| e.to_string())?,
        );
        req
    };

    // 2. 建立连接
    let (ws_stream, _) =
        connect_async(request).await.map_err(|e| e.to_string())?;
    let (mut ws_sink, mut ws_stream) = ws_stream.split();

    // 3. 内部通道：其他线程 → ws_sink
    let (tx, mut rx) = mpsc::unbounded_channel::<Message>();

    // 保存发送端到全局 State
    {
        let mut lock = session_holder.lock().await;
        *lock = Some(OmniSession { tx: tx.clone() });
    }

    // 4. 转发任务：内部通道 → WebSocket
    tokio::spawn(async move {
        while let Some(msg) = rx.recv().await {
            if ws_sink.send(msg).await.is_err() {
                break;
            }
        }
    });

    // 5. 等待 session.created，然后发送 session.update
    let app_clone = app.clone();
    let tx_clone = tx.clone();

    tokio::spawn(async move {
        let mut session_initialized = false;

        while let Some(Ok(msg)) = ws_stream.next().await {
            match msg {
                Message::Text(text) => {
                    let v: Value = match serde_json::from_str(&text) {
                        Ok(v) => v,
                        Err(_) => continue,
                    };

                    match v["type"].as_str().unwrap_or("") {
                        // 5a. 服务端确认连接 → 发送 session.update
                        "session.created" => {
                            if !session_initialized {
                                session_initialized = true;
                                send_session_update(&tx_clone, &system_prompt);
                                let _ = app_clone.emit(
                                    "omni://status",
                                    StatusPayload {
                                        status: "connected".into(),
                                        message: None,
                                    },
                                );
                            }
                        }

                        // 5b. 增量音频 → 发给前端播放
                        "response.audio.delta" => {
                            if let Ok(delta) =
                                serde_json::from_str::<AudioDelta>(&text)
                            {
                                let _ = app_clone.emit(
                                    "omni://audio_chunk",
                                    AudioChunkPayload { data: delta.delta },
                                );
                            }
                        }

                        // 5c. 增量文字 → 发给前端显示
                        "response.text.delta" => {
                            if let Ok(td) =
                                serde_json::from_str::<TextDelta>(&text)
                            {
                                let _ = app_clone.emit(
                                    "omni://text_chunk",
                                    TextChunkPayload { text: td.delta },
                                );
                            }
                        }

                        // 5d. 音频完成
                        "response.audio.done" => {
                            let _ = app_clone.emit(
                                "omni://status",
                                StatusPayload {
                                    status: "audio_done".into(),
                                    message: None,
                                },
                            );
                        }

                        // 5e. 错误
                        "error" => {
                            let msg_str = v["error"]["message"]
                                .as_str()
                                .unwrap_or("unknown error")
                                .to_string();
                            let _ = app_clone.emit(
                                "omni://status",
                                StatusPayload {
                                    status: "error".into(),
                                    message: Some(msg_str),
                                },
                            );
                        }

                        _ => {}
                    }
                }

                Message::Close(_) => {
                    let _ = app_clone.emit(
                        "omni://status",
                        StatusPayload {
                            status: "disconnected".into(),
                            message: None,
                        },
                    );
                    break;
                }

                _ => {}
            }
        }

        // 连接断开后清理 State
    });

    Ok(())
}

// ── 辅助函数 ──────────────────────────────────────────────

fn new_event_id() -> String {
    format!("event_{}", Uuid::new_v4().simple())
}

/// 发送 session.update（握手配置）
fn send_session_update(
    tx: &mpsc::UnboundedSender<Message>,
    system_prompt: &str,
) {
    let msg = json!({
        "event_id": new_event_id(),
        "type": "session.update",
        "session": {
            "modalities": ["text", "audio"],
            "voice": "Tina",                    // Qwen3.5-Omni-Plus-Realtime 默认音色
            "input_audio_format": "pcm",
            "output_audio_format": "pcm",
            "instructions": system_prompt,
            "turn_detection": {
                "type": "server_vad",
                "threshold": 0.5,
                "silence_duration_ms": 800      // VAD 静音超时
            }
        }
    });
    let _ = tx.send(Message::Text(msg.to_string().into()));
}

// ── Tauri Commands ────────────────────────────────────────

/// 前端调用：建立连接
#[tauri::command]
pub async fn omni_connect(
    api_key: String,
    system_prompt: String,
    app: AppHandle,
    session: tauri::State<'_, SharedSession>,
) -> Result<(), String> {
    let holder = session.inner().clone();
    connect(api_key, system_prompt, app, holder).await
}

/// 前端调用：发送一块 PCM 音频（前端传 Base64）
#[tauri::command]
pub async fn omni_send_audio(
    audio_b64: String,
    session: tauri::State<'_, SharedSession>,
) -> Result<(), String> {
    let lock = session.lock().await;
    if let Some(s) = lock.as_ref() {
        let pcm = general_purpose::STANDARD
            .decode(&audio_b64)
            .map_err(|e| e.to_string())?;
        s.append_audio(&pcm);
        Ok(())
    } else {
        Err("未连接".into())
    }
}

/// 前端调用：手动提交音频（Manual 模式）
#[tauri::command]
pub async fn omni_commit_audio(
    session: tauri::State<'_, SharedSession>,
) -> Result<(), String> {
    let lock = session.lock().await;
    if let Some(s) = lock.as_ref() {
        s.commit_audio();
        Ok(())
    } else {
        Err("未连接".into())
    }
}

/// 前端调用：断开连接
#[tauri::command]
pub async fn omni_disconnect(
    session: tauri::State<'_, SharedSession>,
) -> Result<(), String> {
    let mut lock = session.lock().await;
    if let Some(s) = lock.take() {
        s.disconnect();
    }
    Ok(())
}
```

然后在 `main.rs` 注册：

```rust
// src-tauri/src/main.rs
mod qwen_omni;
use qwen_omni::SharedSession;
use std::sync::Arc;
use tokio::sync::Mutex;

fn main() {
    tauri::Builder::default()
        .manage(SharedSession::new(Mutex::new(None))) // 注入 State
        .invoke_handler(tauri::generate_handler![
            qwen_omni::omni_connect,
            qwen_omni::omni_send_audio,
            qwen_omni::omni_commit_audio,
            qwen_omni::omni_disconnect,
        ])
        .run(tauri::generate_context!())
        .expect("error running tauri app");
}
```

React 前端对应的调用示例：

```tsx
import { invoke } from "@tauri-apps/api/core";
import { listen } from "@tauri-apps/api/event";

// 连接
await invoke("omni_connect", {
  apiKey: "sk-xxxx",
  systemPrompt: "你是一个友好的语音助手",
});

// 监听音频块（Base64 PCM，送给 AudioContext 播放）
await listen<{ data: string }>("omni://audio_chunk", ({ payload }) => {
  playPcmChunk(payload.data); // 你的 PCM 解码播放逻辑
});

// 监听文字
await listen<{ text: string }>("omni://text_chunk", ({ payload }) => {
  setTranscript(prev => prev + payload.text);
});

// 监听状态
await listen<{ status: string }>("omni://status", ({ payload }) => {
  console.log("状态:", payload.status);
});

// 发送麦克风 PCM（MediaRecorder 录制后转 Base64）
await invoke("omni_send_audio", { audioB64: base64PcmChunk });
```

**数据流向总结：**

```
麦克风(PCM) → React → [omni_send_audio] → Rust append_audio
                                                    ↓ WebSocket
                                          Qwen3.5-Omni-Plus-Realtime
                                                    ↓ response.audio.delta
Rust → [emit omni://audio_chunk] → React → AudioContext 播放
```
